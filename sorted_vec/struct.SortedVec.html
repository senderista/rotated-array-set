pub struct SortedVec<T> { /* fields omitted */ }

A set based on a 2-level rotated array.

See [README.md] for a detailed discussion of this collection's performance benefits and drawbacks.

Examples

use sorted_vec::SortedVec;

// Type inference lets us omit an explicit type signature (which
// would be `SortedVec<i32>` in this example).
let mut ints = SortedVec::new();

// Add some integers.
ints.insert(-1);
ints.insert(6);
ints.insert(1729);
ints.insert(24);

// Check for a specific one.
if !ints.contains(&42) {
    println!("We don't have the answer to Life, the Universe, and Everything.");
}

// Remove an integer.
ints.remove(&6);

// Iterate over everything.
for int in &ints {
    println!("{}", int);
}
